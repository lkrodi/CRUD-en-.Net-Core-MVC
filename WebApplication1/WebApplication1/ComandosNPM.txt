Comandos para instalar ReactJS, WebPack y todo para el trabajo
Frontend y Backend con ASP .Net Core C#

*********************************
npm init -y
npm i webpack --save-dev
npm i jquery --save
npm i jquery-validation-unobtrusive --save
npm i jquery-validation --save

npm i webpack-hot-middleware --save-dev
npm i aspnet-webpack --save-dev

npm i babel-loader@8.0.0-beta.0 @babel/core@next @babel/preset-env@next --save-dev
npm i babel-loader @babel/core@next @babel/preset-env@next --save
npm i -D babel-loader @babel/core @babel/preset-env webpack

npm i @babel/preset-react --save-dev
npm i react react-dom --save-dev

npm i --dev webpack-dev-middleware

npm run wbp
